#include <bits/stdc++.h>

using namespace std;

typedef long long int ll;
int main() {
    ll n;
    cin>>n;
    ll b[n+1]={0};
    for(ll i=1;i<=n;i++){
        cin>>b[i];
    }
    
    ll q;
    cin>>q;
    while(q--){
        ll w;cin>>w;
        ll vl;cin>>vl; ll u = 0 ; ll p = 0 ; 
        for(ll l=w;l<=n;l++){
            u = (u) | (b[l]) ;
            if(u>=vl){
                p = l ; 
                break;
            }
        }
        
        
        cout<<p;
        cout<<"\n";
    }
    
    
    
    
    
}
